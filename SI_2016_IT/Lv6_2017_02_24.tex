%Om den enda lösningen är x1 = x2 = ... = xn = 0 är de linjärt oberoende

	\item[a) ] $\vec{u} = \begin{bmatrix} 1 \\ 2 \\ 3 \end{bmatrix}$, $\vec{v} = \begin{bmatrix} 3 \\ 1 \\ 5 \end{bmatrix}$, $\vec{w} = \begin{bmatrix} 5 \\ 7 \\ 1 \end{bmatrix}$
