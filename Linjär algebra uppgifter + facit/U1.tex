Låt $\vec{u} = \begin{bmatrix} 1 \\ 3 \end{bmatrix}$ och $\vec{v} = \begin{bmatrix} 2 \\ 5 \end{bmatrix}$. Beräkna följande:
\begin{itemize}
\item[a) ] $\vec{u} + \vec{v}$.
\item[b) ] $\vec{u} - \vec{v}$.
\item[c) ] $3 \cdot \vec{u} + \vec{v}$.
\item[d) ] $2 \cdot(\vec{v} - 2 \cdot \vec{u})$
\item[e) ] $4 \cdot \vec{u} -2 \cdot (\vec{u} -3 \cdot \vec{v})$
\end{itemize}
